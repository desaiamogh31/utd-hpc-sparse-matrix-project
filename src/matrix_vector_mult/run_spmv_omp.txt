#!/bin/bash

# Run OpenMP SpMV benchmarks with different matrix sizes and thread counts
# Similar to nbody_omp timing script - collects runtimes across configurations

set -e

# Define arrays for matrix sizes and thread counts
MATRIX_SIZES=(1000 2000 4000 8000 16000)
THREAD_COUNTS=(1 2 4)

OUTPUT_DIR="results"
OUTPUT_FILE="$OUTPUT_DIR/spmv_openmp_runtimes.txt"

# Create output directory
mkdir -p "$OUTPUT_DIR"
mkdir -p logs

# Clear output file and add header
> "$OUTPUT_FILE"
echo "MatrixSize,Threads,Avg_Time_s,GFlops" >> "$OUTPUT_FILE"

echo "=========================================="
echo "SPMV OpenMP Benchmarking"
echo "=========================================="
echo "Matrix sizes: ${MATRIX_SIZES[@]}"
echo "Thread counts: ${THREAD_COUNTS[@]}"
echo "Output file: $OUTPUT_FILE"
echo "=========================================="

# Build the shared library first
echo "Building OpenMP library..."
python build.py

# Run benchmarks for each configuration
for n in "${MATRIX_SIZES[@]}"; do
    echo ""
    echo "Testing matrix size n=$n..."
    
    for threads in "${THREAD_COUNTS[@]}"; do
        echo "  Running with $threads thread(s)..."
        
        # Run benchmark for this configuration
        # Using NNZ ratio 5.0 (same as default)
        output=$(python -c "
import sys
sys.path.insert(0, '.')
from benchmark_spmv_openmp import benchmark_spmv_omp, generate_test_matrix

A, x = generate_test_matrix($n, 5.0, seed=0)
result = benchmark_spmv_omp(A, x, $threads, repeats=5)
print(f'{result[\"avg_time_s\"]:.6f},{result[\"gflops\"]:.2f}')
")
        
        # Parse output
        avg_time=$(echo "$output" | cut -d',' -f1)
        gflops=$(echo "$output" | cut -d',' -f2)
        
        # Append to results file
        echo "$n,$threads,$avg_time,$gflops" >> "$OUTPUT_FILE"
        
        echo "    Avg time: ${avg_time}s, GFlops: ${gflops}"
    done
done

echo ""
echo "=========================================="
echo "Benchmark complete!"
echo "Results saved to: $OUTPUT_FILE"
echo "=========================================="
echo ""
echo "Summary:"
cat "$OUTPUT_FILE"
